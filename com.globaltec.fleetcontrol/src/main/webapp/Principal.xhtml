function handleMessage(data) { var coords = new google.maps.LatLng(data.lat, data.lng), map = PF('map'), marker = new google.maps.Marker({ position: coords, title: data.title }); map.addOverlay(marker); map.infowindow = new google.maps.InfoWindow({ content: "

" + data.title + " just checked in!

" }); map.getMap().setCenter(coords); map.infowindow.open(map.getMap(), marker); }
